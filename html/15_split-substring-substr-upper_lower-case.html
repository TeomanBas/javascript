split()

bir karakter kümesini istenilen karaktere göre parçalara ayırır ve bu parçaları dizi içerisine yerleştirir

var a="GOP Ünv. turhal MYO";
var dizi_deg=new Array();
dizi_deg=a.split('');//boşluk karakterine göre parçalar.
document.write(dizi_deg);

substring()

bir karakter kümseinden istenilen bir bölümün alınmasını sağlar.birinci parametre başlangıç noktasını ikinci parametre biriş noktasını ifade eder ve bu aralıktaki karakterleri alır

var a="Turhal MYO";
var b=a.substring(0,6);//b değişkeni turhal değeerini alacaktır.

substr()

bir karakter kümesinden istenilen bir bölümün alınmasını sağlar substring fonksiyonundan farkı olarak birinci parametre başlangıç noktasını ikinci parametre karakter sayısını belirtir.

var a="TURHAL MYO";
var b=a.substr(3,3);
document.write(b);//3. karakter dahil H harfi alındı

toUpperCase() ve toLowerCase()

bir karakter kümesindeki elemanları toUpperCase fonksiyonu büyük harfle,toLoverCase fonksiyonu küçük harfe çevirir.

< script type="text/javascript" >
var a ="kaLEMler";

var c =a.toUpperCase();
var d =a.toLowerCase();
document.write(c+" "+d);

< /script >
